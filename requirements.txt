# Python deps for rm-ocr. Everything else it uses is in the standard library
# (urllib, json, hashlib, pathlib, logging, ...).
#
# pdf2image rasterizes PDF pages; it shells out to poppler (pdftoppm / pdfunite),
# which is a SYSTEM package, NOT pip-installable:
#   Debian/Ubuntu : apt-get install poppler-utils
#   macOS         : brew install poppler
# pdf2image pulls in Pillow automatically.
pdf2image>=1.17

# pypdf reads the splitter's /RemarkableSplitter metadata marker and page sizes
# for the split-readiness gate (REQUIRE_SPLIT=1) and the in-place splitter
# (AUTO_SPLIT=1). Lazily imported, so only strictly required when a split feature
# is enabled.
pypdf>=6.14.2

# numpy + Pillow are used ONLY by AUTO_SPLIT (rm_split.py: whitespace-band
# detection on rendered pages). Pillow is already pulled in by pdf2image; numpy is
# the only addition. Both lazily imported — not needed unless AUTO_SPLIT=1.
numpy>=2.5.0
Pillow>=12.2.0

# inotify_simple powers the optional inotify wake-up signal (INOTIFY=1, on by
# default). Linux-only — on macOS the import will fail and the daemon falls back
# to pure polling. Pure-stdlib wrapper, no transitive deps.
inotify_simple>=2.0.1 ; sys_platform == "linux"

# rmc renders reMarkable .zip / .rmdoc / .rm bundles to PDF. Pulls in rmscene
# (the .rm parser by the same author) transitively. Invoked with --no-chrome so
# we don't need Chrome or cairo in the image — pdfunite (poppler-utils, already
# above) is the only system dep on the bundle path.
rmc>=0.3
